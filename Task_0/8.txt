#include <iostream>
#include <stdlib.h>

using namespace std;

struct my_stuct
{
	char c;

	my_stuct() : c('a'), a(0), b(5.0) {}

	void f()
	{
		cout << "i said yyyy iam blinding by the light" << endl;
	}

template<typename T>
T* my_new()
{
	void* pointer = malloc(sizeof(T));

	if (malloc == NULL) {
		return NULL;
	}

	T* pobj = new(pointer) T();
	return pobj;
}

{

	a->f();
